/**
 * Checks whether type has a correct {datatype}
 * @name isValidDataType
 * @param {mixed} type
 * @return {bool}
 */
function isValidDataType(datatype) {
    return datatype == "node" || datatype == "int64" || datatype == "string" || datatype == "float" || datatype == "bool" || datatype == "timestamp";
}
